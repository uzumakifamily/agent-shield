Wrap any action with shield.run(). The action only executes if the verdict is ALLOW.

Install + usage
npm install agent-shield-core

const shield = require('agent-shield-core');

const result = await shield.run(
  {
    workspaceId: 'YOUR_WORKSPACE_ID',
    projectId:   'my-project',
    agent:       'my-agent',
    actionType:  'production_deploy',
    source:      'api',
    payload:     { service: 'api', env: 'prod' }
  },
  async () => deployToProduction()  // only runs if ALLOW
);

if (!result.proceed) {
  // verdict is 'PAUSE' or 'BLOCK'
  console.log(result.verdict, result.reason);
}

No SDK needed — call the REST API with requests.

Python helper + example
import requests

API_KEY = "YOUR_API_KEY"
BASE    = "https://zippy-spontaneity-production.up.railway.app"

def check_action(action_type, payload, agent="my-agent"):
    r = requests.post(f"{BASE}/api/actions",
        headers={"Authorization": f"Bearer {API_KEY}",
                 "Content-Type": "application/json"},
        json={"action_type": action_type, "payload": payload,
              "context": {"agent": agent, "source": "api"}})
    return r.json()

result = check_action("file_delete", {"path": "/data/users.csv"})

if   result["verdict"] == "ALLOW": delete_file("/data/users.csv")
elif result["verdict"] == "PAUSE": print("Waiting:", result["approval_id"])
elif result["verdict"] == "BLOCK": print("Blocked:", result["reason"])
